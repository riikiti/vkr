## 7.1 Назначение модуля

Модуль дополняет энтропийный анализ классическими статистическими тестами. Если энтропийные метрики показывают «глобальную» случайность, то статистические тесты выявляют локальные структуры и аномалии в распределении байтов шифртекста.

**Зависимости:** `numpy`, `scipy`

---

## 7.2 `frequency_analysis.py` — Частотный анализ байтов

### Назначение
Строит гистограмму частот каждого из 256 возможных байтов. Для идеального шифра распределение должно быть равномерным — каждый байт встречается с частотой `1/256`.

```python
# core/statistics/frequency_analysis.py
import numpy as np
from scipy.stats import chisquare

def get_byte_frequencies(data: bytes) -> np.ndarray:
    """
    Возвращает массив длиной 256: количество вхождений каждого байта.
    """
    return np.bincount(np.frombuffer(data, dtype=np.uint8), minlength=256)


def get_byte_probabilities(data: bytes) -> np.ndarray:
    """
    Возвращает нормализованное распределение вероятностей (сумма = 1).
    """
    counts = get_byte_frequencies(data)
    return counts / len(data)


def chi_square_uniformity_test(data: bytes) -> dict:
    """
    Тест хи-квадрат на равномерность распределения байтов.

    Нулевая гипотеза H0: распределение равномерное.
    Если p-value > 0.05 → не отвергаем H0 (распределение близко к равномерному).

    Возвращает:
        chi2_stat: статистика хи-квадрат
        p_value:   p-значение
        is_uniform: True если p_value > 0.05
    """
    counts = get_byte_frequencies(data)
    expected = np.full(256, len(data) / 256)
    chi2, p_value = chisquare(counts, f_exp=expected)
    return {
        "chi2_stat": float(chi2),
        "p_value": float(p_value),
        "is_uniform": bool(p_value > 0.05),
    }


def frequency_deviation(data: bytes) -> float:
    """
    Среднее абсолютное отклонение частот от идеального равномерного распределения.
    Меньше → лучше.
    """
    probs = get_byte_probabilities(data)
    ideal = 1 / 256
    return float(np.mean(np.abs(probs - ideal)))
```

## 7.3 `correlation.py` — Корреляционный анализ

### Назначение
Проверяет статистическую зависимость между соседними байтами шифртекста (автокорреляция) и между байтами открытого и зашифрованного текста.

```python
# core/statistics/correlation.py
import numpy as np
from scipy.stats import pearsonr, spearmanr

def calculate_autocorrelation(data: bytes, lag: int = 1) -> float:
    """
    Автокорреляция байтов с задержкой lag.
    Значение близкое к 0 → нет линейной зависимости между соседними байтами.
    Для идеального шифра: ≈ 0.
    """
    arr = np.frombuffer(data, dtype=np.uint8).astype(float)
    x = arr[:-lag]
    y = arr[lag:]
    if len(x) < 2:
        return 0.0
    corr, _ = pearsonr(x, y)
    return float(corr)


def calculate_plaintext_ciphertext_correlation(
    plaintext: bytes,
    ciphertext: bytes
) -> dict:
    """
    Корреляция Пирсона и Спирмена между открытым и зашифрованным текстом.
    Для идеального шифра оба значения ≈ 0.
    """
    min_len = min(len(plaintext), len(ciphertext))
    if min_len < 2:
        return {"pearson": 0.0, "spearman": 0.0}

    x = np.frombuffer(plaintext[:min_len], dtype=np.uint8).astype(float)
    y = np.frombuffer(ciphertext[:min_len], dtype=np.uint8).astype(float)

    pearson_r, _ = pearsonr(x, y)
    spearman_r, _ = spearmanr(x, y)

    return {
        "pearson": float(pearson_r),
        "spearman": float(spearman_r),
    }


def calculate_autocorrelation_profile(data: bytes, max_lag: int = 10) -> list[float]:
    """
    Профиль автокорреляции для lag = 1..max_lag.
    Позволяет построить ACF-график.
    """
    return [calculate_autocorrelation(data, lag=i) for i in range(1, max_lag + 1)]
```

## 7.4 `distribution_metrics.py` — Метрики распределения

### Назначение
Статистические моменты распределения байтов: дисперсия, асимметрия, эксцесс. Для равномерного распределения [0, 255] ожидаемые значения известны, что даёт точные критерии сравнения.

```python
# core/statistics/distribution_metrics.py
import numpy as np
from scipy.stats import skew, kurtosis

# Эталонные значения для равномерного распределения [0, 255]
UNIFORM_MEAN = 127.5
UNIFORM_VAR = 5461.25       # (256^2 - 1) / 12
UNIFORM_STD = 73.9          # sqrt(UNIFORM_VAR)
UNIFORM_SKEWNESS = 0.0      # симметричное
UNIFORM_KURTOSIS = -1.2     # эксцесс равномерного распределения

def calculate_distribution_metrics(data: bytes) -> dict:
    """
    Вычисляет статистические моменты распределения байтов.

    Возвращает:
        mean:      среднее (для равномерного: 127.5)
        variance:  дисперсия (для равномерного: ≈ 5461)
        std:       СКО (для равномерного: ≈ 73.9)
        skewness:  асимметрия (для равномерного: ≈ 0)
        kurtosis:  эксцесс (для равномерного: ≈ -1.2)
    """
    arr = np.frombuffer(data, dtype=np.uint8).astype(float)

    return {
        "mean": float(np.mean(arr)),
        "variance": float(np.var(arr)),
        "std": float(np.std(arr)),
        "skewness": float(skew(arr)),
        "kurtosis": float(kurtosis(arr)),  # excess kurtosis (Fisher)
    }


def compare_to_uniform(metrics: dict) -> dict:
    """
    Сравнивает метрики с эталонными значениями равномерного распределения.
    Возвращает отклонения (абсолютные).
    """
    return {
        "mean_deviation": abs(metrics["mean"] - UNIFORM_MEAN),
        "variance_deviation": abs(metrics["variance"] - UNIFORM_VAR),
        "skewness_deviation": abs(metrics["skewness"] - UNIFORM_SKEWNESS),
        "kurtosis_deviation": abs(metrics["kurtosis"] - UNIFORM_KURTOSIS),
    }


def calculate_max_index_of_coincidence(data: bytes) -> float:
    """
    Индекс совпадений — дополнительная метрика из классического
    криптоанализа. Для случайных данных ≈ 0.00390 (1/256).
    Для английского текста ≈ 0.065.
    """
    counts = np.bincount(np.frombuffer(data, dtype=np.uint8), minlength=256)
    n = len(data)
    if n < 2:
        return 0.0
    ic = np.sum(counts * (counts - 1)) / (n * (n - 1))
    return float(ic)
```

## 7.5 Единая точка входа `__init__.py`

```python
# core/statistics/__init__.py
from .frequency_analysis import (
    get_byte_frequencies,
    get_byte_probabilities,
    chi_square_uniformity_test,
    frequency_deviation,
)
from .correlation import (
    calculate_autocorrelation,
    calculate_plaintext_ciphertext_correlation,
    calculate_autocorrelation_profile,
)
from .distribution_metrics import (
    calculate_distribution_metrics,
    compare_to_uniform,
    calculate_max_index_of_coincidence,
)

def analyze_all_statistics(plaintext: bytes, ciphertext: bytes) -> dict:
    """Запускает полный статистический анализ шифртекста."""
    freq = chi_square_uniformity_test(ciphertext)
    dist = calculate_distribution_metrics(ciphertext)
    corr = calculate_plaintext_ciphertext_correlation(plaintext, ciphertext)
    return {
        **{f"freq_{k}": v for k, v in freq.items()},
        **{f"dist_{k}": v for k, v in dist.items()},
        **{f"corr_{k}": v for k, v in corr.items()},
        "autocorr_lag1": calculate_autocorrelation(ciphertext, lag=1),
        "freq_deviation": frequency_deviation(ciphertext),
        "index_of_coincidence": calculate_max_index_of_coincidence(ciphertext),
    }
```

## 7.6 Интерпретация результатов

| Метрика | Идеальное значение | Интерпретация |
|---------|-------------------|---------------|
| χ² p-value | > 0.05 | Распределение не отличается от равномерного |
| Pearson корреляция | ≈ 0 | Нет линейной зависимости |
| Асимметрия | ≈ 0 | Симметричное распределение |
| Эксцесс | ≈ −1.2 | Равномерное распределение |
| Индекс совпадений | ≈ 0.00390 | Случайное распределение |
